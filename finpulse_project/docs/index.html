// ---- Metrics Tab ---- const METRICS_URL = 'https://finpulse-intelligence-c819.onrender.com/observability/metrics'; let metricsToolChart = null; let metricsCacheChart = null; function parsePrometheusMetric(text, metricName) { const lines = text.split('\n').filter(l => l.startsWith(metricName) && !l.startsWith('#')); const result = {}; lines.forEach(line => { const labelMatch = line.match(/\{([^}]+)\}/); const valueMatch = line.match(/\}\s+([\d.e+-]+)/) || line.match(/\s+([\d.e+-]+)$/); if (valueMatch) { const key = labelMatch ? labelMatch[1] : 'value'; result[key] = parseFloat(valueMatch[1]); } }); return result; } async function loadMetrics() { try { const resp = await fetch(METRICS_URL); if (!resp.ok) throw new Error('HTTP ' + resp.status); const text = await resp.text(); // Show raw metrics (FinPulse-specific lines only) const finpulseLines = text.split('\n').filter(l => l.includes('finpulse_') || l.startsWith('# HELP finpulse') || l.startsWith('# TYPE finpulse')); document.getElementById('m-raw').textContent = finpulseLines.join('\n') || 'No finpulse_ metrics yet — make a few API requests first.'; // Parse tool executions const toolData = parsePrometheusMetric(text, 'finpulse_tool_executions_total'); const toolSuccess = {}, toolError = {}; Object.entries(toolData).forEach(([labels, val]) => { const nameMatch = labels.match(/tool_name="([^"]+)"/); const outcomeMatch = labels.match(/outcome="([^"]+)"/); if (nameMatch && outcomeMatch) { const name = nameMatch[1].replace('get_', ''); if (outcomeMatch[1] === 'success') toolSuccess[name] = val; else if (outcomeMatch[1] === 'error') toolError[name] = val; } }); const toolNames = [...new Set([...Object.keys(toolSuccess), ...Object.keys(toolError)])]; const totalTools = Object.values(toolSuccess).reduce((a,b)=>a+b,0) + Object.values(toolError).reduce((a,b)=>a+b,0); document.getElementById('m-tools').textContent = Math.round(totalTools); document.getElementById('m-tool-errors').textContent = Object.values(toolError).reduce((a,b)=>a+b,0) + ' errors'; // Tool chart if (metricsToolChart) metricsToolChart.destroy(); metricsToolChart = new Chart(document.getElementById('metricsToolChart'), { type: 'bar', data: { labels: toolNames.length ? toolNames : ['no data yet'], datasets: [ { label: 'Success', data: toolNames.map(n => toolSuccess[n] || 0), backgroundColor: '#00d4aa88', borderColor: '#00d4aa', borderWidth: 1 }, { label: 'Error', data: toolNames.map(n => toolError[n] || 0), backgroundColor: '#f8717188', borderColor: '#f87171', borderWidth: 1 } ] }, options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { labels: { color: '#94a3b8' } } }, scales: { x: { grid: { color: '#1e2a3a' }, ticks: { color: '#94a3b8' } }, y: { grid: { color: '#1e2a3a' }, ticks: { color: '#94a3b8' }, beginAtZero: true } } } }); // Parse cache hits const cacheData = parsePrometheusMetric(text, 'finpulse_cache_hits_total'); let hits = 0, misses = 0; Object.entries(cacheData).forEach(([labels, val]) => { if (labels.includes('result="hit"')) hits = val; if (labels.includes('result="miss"')) misses = val; }); const total = hits + misses; const hitRate = total > 0 ? ((hits / total) * 100).toFixed(1) : 0; document.getElementById('m-cache-rate').textContent = hitRate + '%'; document.getElementById('m-cache-abs').textContent = hits + ' hits / ' + misses + ' misses'; // Cache chart if (metricsCacheChart) metricsCacheChart.destroy(); metricsCacheChart = new Chart(document.getElementById('metricsCacheChart'), { type: 'doughnut', data: { labels: ['Cache Hits', 'Cache Misses'], datasets: [{ data: [hits || 0, misses || 0], backgroundColor: ['#00d4aa', '#f87171'], borderWidth: 0, hoverOffset: 6 }] }, options: { responsive: true, maintainAspectRatio: false, cutout: '65%', plugins: { legend: { labels: { color: '#94a3b8' } } } } }); // Total requests const reqData = parsePrometheusMetric(text, 'finpulse_http_requests_total'); const totalReqs = Object.values(reqData).reduce((a,b)=>a+b,0); document.getElementById('m-total-req').textContent = Math.round(totalReqs); document.getElementById('m-req-rate').textContent = 'since last restart'; } catch(e) { document.getElementById('m-raw').textContent = 'Could not load metrics — server may be waking up (30-60s). Try again shortly.\n\nError: ' + e.message; } } // Auto-load metrics when Metrics tab is clicked const origShowPage = showPage; window.showPage = function(id) { origShowPage(id); if (id === 'metrics' && !metricsToolChart) loadMetrics(); };
